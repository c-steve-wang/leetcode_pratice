class Solution:
    def rob(self, nums: List[int]) -> int:
        # take current point or not
        # dp[i] max money take

        n = len(nums)
        dp = [0]* (n + 1)
        dp[0] = nums[0]

        for i in range(1, n):
            dp[i] = max(dp[i-2] + nums[i], dp[i - 1])


        return dp[n - 1]
